## iOS
```Swift
public protocol SesameProtocolMechStatus {
    var data: Data { get }
    var position: Int16 { get }        // セサミデバイスのリアルタイムの角度。範囲： -32767~0~32767 ; -32768 は意義のないデフォルト值。 例0˚ ⇄ 0 で 360˚ ⇄ 1024 
    var target: Int16 { get }
    var isClutchFailed: Bool { get }   // 施錠範囲であるか否か の判断   
    var isInLockRange: Bool{ get }
    var isInUnlockRange: Bool { get } // 解錠範囲であるか否か の判断
    func getBatteryVoltage() -> Float  // 電池の電圧の取得、単位: 0V ~ 7.2V
    func getBatteryPrecentage() -> Int
    func retCodeType() -> Sesame2RetCodeType
}
```
